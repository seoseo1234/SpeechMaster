<script>
        let isRecording = false;
        function toggleRecording() {
            isRecording = !isRecording;
            const statusDiv = document.getElementById('recording-status');
            const micText = document.getElementById('mic-text');
            const micIcon = document.getElementById('mic-icon');
            const button = document.querySelector('.pulse-mic');

            if (isRecording) {
                statusDiv.classList.remove('hidden');
                micText.innerText = '말하는 중...';
                micIcon.innerText = 'stop';
                button.classList.replace('chunky-button-secondary', 'chunky-button-primary');
                button.style.backgroundColor = '#ba1a1a'; // Red stop state
                button.style.boxShadow = '0 6px 0 0 #93000a';
            } else {
                statusDiv.classList.add('hidden');
                micText.innerText = '누르고 말하기';
                micIcon.innerText = 'mic';
                button.style.backgroundColor = ''; // Restore to Sky Blue from CSS
                button.style.boxShadow = '';
                button.classList.replace('chunky-button-primary', 'chunky-button-secondary');
            }
        }
    </script>
